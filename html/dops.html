function DigPow(n, k) {
    let arr = [];
    let counter = 0;
    arr.push(n);
    arr = arr.join('').split('');
    arr.forEach((el, i) => counter += Math.pow(el, k + i));
    return counter % n === 0 ? counter / n : -1;
}
console.log(DigPow(89, 1));
console.log(DigPow(695, 2));

function Wave(str) {
    let ResultArray = [];
    for (let i = 0; i < str.length; i++) {
        if (str[i] !== ' ') {
            let NewStr = str.slice(0, i) + str[i].toUpperCase() + str.slice(i + 1);
            ResultArray.push(NewStr);
        }
    }
    return ResultArray;
}
console.log(Wave("hello"));
console.log(Wave("gap"));

function ConvertString(str) {
    return str.toLowerCase().split('').map(el =>
        str.indexOf(el) === str.lastIndexOf(el) ? '(' : ')'
    ).join('');
}
console.log(ConvertString('din'));
console.log(ConvertString('recede'));

let AlphToLower = 'abcdefghijklmnopqrstuvwxyz'
let AlphToUpper = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'

function LostLetter(arr) {
	let FirstIndex = AlphToLower.indexOf(arr[0])
	let FirstIndexUpper = AlphToUpper.indexOf(arr[0])

	if (arr.every(el => el === el.toLowerCase())) {
		for (let i = 0; i < arr.length; i++) {
			if (AlphToLower[FirstIndex + i] !== arr[i]) {
				return AlphToLower[FirstIndex + i]
			}
		}
		return AlphToLower[FirstIndex + arr.length];
	}
	else if (arr.every(el => el === el.toUpperCase())) {
		for (let i = 0; i < arr.length; i++) {
		 if (AlphToUpper[FirstIndexUpper + i] !== arr[i]) {
				return AlphToUpper[FirstIndexUpper + i]
			}
		}
return AlphToUpper[FirstIndexUpper + arr.length]
	}
}
console.log(LostLetter(['a', 'b', 'c', 'd']));  
console.log(LostLetter(['O', 'Q', 'R', 'S']));

]
function FindPos(maze) {
for (let i = 0; i < maze.length; i++) {
for (let j = 0; j < maze[i].length; j++) {
	if (maze[i][j] == 2) return [i, j]
}
}
}
function mazeRunner(maze, directions) {
let pos = FindPos(maze)
for (let i = 0; i < directions.length; i++) {
switch(directions[i]) {
case 'N':
	pos[0] -= 1
	break
case "S":
	pos[0] += 1
	break
case 'E':
	pos[1] += 1
	break
case 'W':
	pos[1] -= 1
	break
}
if (pos[0] < 0 || pos[0] >= maze.length || pos[1] < 0 || pos[1] >= maze[0].length) {
	 return 'деад'
}
let FinalPos = maze[pos[0]][pos[1]]
if (FinalPos == 1) return 'вы в сткене'
if (FinalPos == 3) return 'ура'
}
return 'лост'
}

function CheckBrackets(str) {
let count = 0

for (let i of str) {
	if (i == '(') count++
	if (i == ')') count--
	if (count < 0) return false
}
if ( count == 0 ) return count == 0
}

console.log(CheckBrackets('()()()())')); // false
console.log(CheckBrackets('((((()))))')); // true
console.log(CheckBrackets('(())((()((()))))')); // true
console.log(CheckBrackets('()')); // true
console.log(CheckBrackets('()()')); // true
console.log(CheckBrackets('))((')); // false
console.log(CheckBrackets('(())((()((()))))')); // true

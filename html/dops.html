	function DigPow(n, k) {
		let arr = []
		let counter = 0
		arr.push(n)
		arr = arr.join('').split('') 
		arr.forEach((el,i) => counter += Math.pow(el, k+i))
		if (counter % n == 0) return counter / n
		else return -1 
	}
	console.log(DigPow(89,1))
	console.log(DigPow(695,2))
	console.log(DigPow(92,1))
	console.log(DigPow(46288,3))

	function Wave(str) {
		let ResultArray = []
		let NewStr = ''
		for (let i = 0; i < str.length; i++) {
			if (str[i] != ' ') {
				NewStr = str.slice(0,i) + str[i].toUpperCase() + str.slice(i+1)
				ResultArray.push(NewStr)
   }
  }
		return ResultArray
	}

function ConvertString(str) { 
      return str.toLowerCase().split('').map(el => str.indexOf(el) == str.lastIndexOf(el) ? '(' : ')').join('')
}

console.log(ConvertString('din'));
console.log(ConvertString('recede'));
console.log(ConvertString('Success'));
console.log(ConvertString('(( @'));
